        function updateSyncStatus(type) {

            switch(type) {

                case 'connecting': 

                    statusEl.className = "sync-indicator bg-slate-100 text-slate-500";

                    statusText.innerText = t('syncing'); 

                    break;

                case 'saving': 

                    statusEl.className = "sync-indicator bg-amber-50 text-amber-600";

            }

        }

        function getGroupedWeeks() {

            const start = parseLocalDate(currentStart);

            const end = parseLocalDate(currentEnd);

            if (isNaN(start.getTime()) || isNaN(end.getTime()) || start > end) return [];

            let current = new Date(start);

            const weeks = [];

            let activeWeek = [];

            while (current <= end) {

                const y = current.getFullYear();

                const m = String(current.getMonth() + 1).padStart(2, '0');

                const d = String(current.getDate()).padStart(2, '0');

                const dateKey = `${y}-${m}-${d}`;


                activeWeek.push({

                    date: new Date(current), key: dateKey, dayOfWeek: current.getDay(),

                    isHoliday: redDays.includes(dateKey) || current.getDay() === 0,

                    isWeekend: current.getDay() === 0 || current.getDay() === 6,

                    holidayName: holidays[dateKey], status: state[dateKey] || 0

                });

                if (current.getDay() === 0 || current.getTime() >= end.getTime()) {

                    weeks.push([...activeWeek]);

                    activeWeek = [];

                }

                current.setDate(current.getDate() + 1);

            }

            return weeks;

        }


        function render() {

            const content = document.getElementById('calendar-content');

            const timestamp = document.getElementById('export-timestamp');

            if (!content) return;

            content.innerHTML = '';

            if (timestamp) timestamp.innerText = `${t('updated')}: ${new Date().toLocaleTimeString(currentLang === 'sv' ? 'sv-SE' : 'en-US')}`;

            

            const weeks = getGroupedWeeks();

            weeks.forEach((week) => {

                const weekNum = getWeekNumber(week[0].date);

                const wrapper = document.createElement('div');

                wrapper.innerHTML = `<div class="flex items-center gap-2 mb-2"><span class="text-[9px] font-black text-slate-300 uppercase tracking-widest">${t('week')}${weekNum}</span><div class="h-px bg-slate-100 flex-grow"></div></div>`;

                const grid = document.createElement('div'); 

                grid.className = "calendar-row";

                

                for (let col = 1; col <= 7; col++) {

                    const day = week.find(d => {

                        const dayNum = d.date.getDay() === 0 ? 7 : d.date.getDay();

                        return dayNum === col;

                    });

                    const card = document.createElement('div');

                    card.style.gridColumnStart = col;


                    if (day) {

                        card.className = `day-card p-2 md:p-3 rounded-xl border flex flex-col justify-between h-20 md:h-24 status-${day.status} ${day.isWeekend ? 'weekend-card' : 'shadow-sm'}`;

                        card.onclick = async () => {

                            state[day.key] = (day.status + 1) % 3;

                            await saveSelection();

                            render();

                        };

                        card.innerHTML = `<div class="flex justify-between items-start">

                            <div class="leading-none overflow-hidden">

                                <div class="text-[7px] md:text-[8px] font-bold text-slate-400 uppercase mb-0.5">${t('days')[day.dayOfWeek]}</div>

                                <div class="${day.isWeekend ? 'text-xs md:text-sm' : 'text-sm md:text-lg'} font-black ${day.isHoliday ? 'red-day-text' : 'text-slate-800'} whitespace-nowrap">

                                    ${day.date.getDate()} <span class="text-[7px] md:text-[8px] font-medium opacity-40">${t('months')[day.date.getMonth()].toUpperCase()}</span>

                                </div>

                            </div>

                        </div>

                        <div class="mt-auto">

                            ${day.holidayName ? `<div class="text-[6px] md:text-[7px] font-black text-red-500 uppercase truncate">${t(day.holidayName)}</div>` : ''}

                            <div class="status-text text-[7px] md:text-[9px] font-black ${day.status === 1 ? 'text-emerald-600' : day.status === 2 ? 'text-red-600' : 'text-slate-200'}">

                                ${day.status === 1 ? t('available') : day.status === 2 ? t('busy') : ''}

                            </div>

                        </div>`;

                    } else {

                        card.className = "ghost-card p-2 md:p-3 rounded-xl h-20 md:h-24";

                    }

                    grid.appendChild(card);

                }

                wrapper.appendChild(grid); 

                content.appendChild(wrapper);

            });

        }


        function getWeekNumber(d) {

            d = new Date(Date.UTC(d.getFullYear(), d.getMonth(), d.getDate()));

            d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay() || 7));

            return Math.ceil((((d - new Date(Date.UTC(d.getUTCFullYear(), 0, 1))) / 86400000) + 1) / 7);

        }
